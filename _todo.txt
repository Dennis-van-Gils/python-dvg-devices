and in RS232 protocol module:
    
    def close(self, ignore_exceptions=False):
        """Close the serial port
        """
        if self.is_alive and self.ser is not None:
            try:
                self.ser.close()
                self.is_alive = False
            except:
                if ignore_exceptions:
                    pass
                else:
                    raise
